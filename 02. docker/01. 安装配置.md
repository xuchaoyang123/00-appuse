
## 一 、安装测试

#
### 1. 清理环境
```shell

                                                                    #清理环境
yum remove docker \                             
                  docker-client \
                  docker-client-latest \
                  docker-common \
                  docker-latest \
                  docker-latest-logrotate \
                  docker-logrotate \
                  docker-engine


```

### 2. yum安装配置
```shell

                                                                    #配置yum源地址
yum-config-manager \
    --add-repo \
    https://download.docker.com/linux/centos/docker-ce.repo

yum install docker-ce docker-ce-cli containerd.io -y               安装docker ce 社区版


root@zhengzelin:~# cat /etc/docker/daemon.json                      #更换国内镜像站
    #网易163镜像加速
    {
    "registry-mirrors": ["http://hub-mirror.c.163.com"]
    }

```

### 3. 重启服务进行测试

```shell
root@zhengzelin:~# sudo systemctl daemon-reload                     #重启服务
root@zhengzelin:~# sudo systemctl restart docker                    #重启服务


docker version                                                      #查看版本!
docker  info                                                        #详细的信息 包括镜像和容器个数
docker  --help

 
 docker run  hello-word                                             #测试

```
